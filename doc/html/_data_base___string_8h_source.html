591  const std::string& v) {
592  auto k_str = r_strings_.find(k) ;
593 
594  // Padding with blank spaces instead of null characters
595  // Key does not exist
596  if ( k_str == r_strings_.end() ) {
597  std::string padding (offset, ' ') ;
598  r_strings_[k] = padding + v ;
599  return ( r_strings_[k].size() ) ;
600  }
601  // Key exists
602  else {
603  auto k_str_size = k_str->second.size() ;
604  if ( offset > k_str->second.size() ) {
605  std::string padding (offset - k_str_size, ' ') ;
606  k_str->second += padding ;
607  }
608  k_str->second.replace(offset, v.size(), v) ;
609  return ( k_str->second.size() ) ;
